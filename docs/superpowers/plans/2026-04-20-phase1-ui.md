## Key Constants & Formulas

These values are used across tasks — read this before implementing any component.

| Concept | Value |
|---|---|
| Cooldown duration | 5000 ms |
| Cooldown tick interval | 100 ms |
| Urgency threshold | < 15 seconds remaining |
| Effective throughput | `Math.max(100, node.throughput * (1 − node.heat / 100))` |
| Critical threshold | `node.heat >= 80` → `node.status === 'critical'` |
| ASCII bar | 10 chars, `█` filled, `░` empty |

Upgrade effects in `gameState.ts` are additive flat values (not multipliers):
- `upgrade-throughput-1`: `+50 throughput` (costs 100¢)
- `upgrade-throughput-2`: `+100 throughput` (costs 200¢)
- `upgrade-efficiency-1`: `+0.1 efficiency` (costs 150¢)
- `upgrade-cooling-20`: `+0.5 efficiency` (costs 120¢)
- `upgrade-cooling-beta`: `+0.4 efficiency` (costs 250¢)

```typescript
coolNode(nodeId: string): void {
  const nodes = this.nodeManager.getState().nodes
  const node = nodes.find(n => n.id === nodeId)
  if (!node) return
  node.heat = Math.max(0, node.heat - 30)
}

purchaseUpgrade(upgradeId: string, nodeId: string): boolean {
  const nodes = this.nodeManager.getState().nodes
  const node = nodes.find(n => n.id === nodeId)
  if (!node) return false
  if (node.upgrades.includes(upgradeId)) return false

  const upgrade = this.state.upgrades.find(u => u.id === upgradeId)
  if (!upgrade) return false

  if (this.economy.getState().credits < upgrade.cost) return false

  this.economy.subtractCredits(upgrade.cost)
  node.upgrades.push(upgradeId)
  // Pass only the new upgrade ID — re-applying all upgrades would double-stack stats
  this.upgradeSystem.applyUpgradesToNode(node, [upgradeId])
  return true
}
```

```typescript
import { create } from 'zustand'
import { GameEngine } from '@/lib/engine/gameEngine'
import { GameState } from '@/lib/engine/types'
import { createInitialGameState } from '@/lib/engine/gameState'

interface GameStore {
  engine: GameEngine | null
  state: GameState | null
  initializeGame: () => void
  tick: () => void
  coolNode: (nodeId: string) => void
  purchaseUpgrade: (upgradeId: string, nodeId: string) => void
}

export const useGameStore = create<GameStore>((set, get) => ({
  engine: null,
  state: null,

  initializeGame: () => {
    const initialState = createInitialGameState()
    const engine = new GameEngine(initialState)
    set({ engine, state: engine.getState() })
  },

  tick: () => {
    const { engine } = get()
    if (!engine) return
    engine.tick()
    set({ state: engine.getState() })
  },

  coolNode: (nodeId: string) => {
    const { engine } = get()
    if (!engine) return
    engine.coolNode(nodeId)
    set({ state: engine.getState() })
  },

  purchaseUpgrade: (upgradeId: string, nodeId: string) => {
    const { engine } = get()
    if (!engine) return
    engine.purchaseUpgrade(upgradeId, nodeId)
    set({ state: engine.getState() })
  },
}))
```

```typescript
export function asciiBar(value: number, max: number, length = 10): string {
  const ratio = Math.min(1, Math.max(0, value / max))
  const filled = Math.round(ratio * length)
  return '█'.repeat(filled) + '░'.repeat(length - filled)
}
```

### Task 4: `NodeCard` Component

**Files:**
- Create: `src/components/game/NodeCard.tsx`

The COOL button has two states: **ready** (green, clickable) and **on cooldown** (gray, disabled with ASCII drain bar). The drain bar is managed with local `cooldownRemaining` state (ms), decremented every 100ms via `setTimeout`.

- [ ] **Step 1: Create `src/components/game/NodeCard.tsx`**

```typescript
'use client'

import { useState, useEffect } from 'react'
import { Node } from '@/lib/engine/types'
import { asciiBar } from '@/lib/asciiBar'

interface NodeCardProps {
  node: Node
  onCool: () => void
}

function getEffectiveThroughput(node: Node): number {
  return Math.max(100, node.throughput * (1 - node.heat / 100))
}

export function NodeCard({ node, onCool }: NodeCardProps) {
  const [cooldownRemaining, setCooldownRemaining] = useState(0)

  // Drain the cooldown bar 10 times per second
  useEffect(() => {
    if (cooldownRemaining <= 0) return
    const timer = setTimeout(
      () => setCooldownRemaining(prev => Math.max(0, prev - 100)),
      100
    )
    return () => clearTimeout(timer)
  }, [cooldownRemaining])

  const handleCool = () => {
    if (cooldownRemaining > 0) return
    onCool()
    setCooldownRemaining(5000)
  }

  const isCritical = node.status === 'critical'
  const borderColor = isCritical ? 'border-red-500' : 'border-green-400'
  const labelColor = isCritical ? 'text-red-400' : 'text-green-400'
  const statusLabel = isCritical ? '● CRITICAL' : '● ONLINE'

  const effective = getEffectiveThroughput(node)

  return (
    <div className={`border ${borderColor} p-4 font-mono text-sm space-y-2`}>
      <div className="flex justify-between">
        <span className={labelColor}>▶ {node.id.toUpperCase()}</span>
        <span className={labelColor}>{statusLabel}</span>
      </div>

      <div className={labelColor}>
        THROUGHPUT {asciiBar(effective, node.throughput)} {effective} pps
      </div>
      <div className={labelColor}>
        HEAT &nbsp;&nbsp;&nbsp;&nbsp;&nbsp; {asciiBar(node.heat, 100)} {Math.round(node.heat)}°
      </div>

      <div className="pt-1">
        {cooldownRemaining > 0 ? (
          <span className="text-gray-600 border border-gray-700 px-3 py-1">
            [ COOL ] {asciiBar(cooldownRemaining, 5000)}
          </span>
        ) : (
          <button
            onClick={handleCool}
            className="border border-green-400 text-green-400 px-3 py-1 hover:bg-green-400 hover:text-black transition-colors"
          >
            [ COOL ]
          </button>
        )}
      </div>
    </div>
  )
}
```

```typescript
import { Contract } from '@/lib/engine/types'
import { asciiBar } from '@/lib/asciiBar'

interface ContractListProps {
  contracts: Contract[]
}

function getTimeRemaining(deadline: number): number {
  return Math.max(0, Math.ceil((deadline - Date.now()) / 1000))
}

export function ContractList({ contracts }: ContractListProps) {
  const active = contracts.filter(c => c.status === 'active')

  if (active.length === 0) {
    return (
      <div className="border border-yellow-700 p-4 font-mono text-sm text-yellow-700">
        <div className="text-xs mb-1">ACTIVE CONTRACTS</div>
        NO ACTIVE CONTRACTS — waiting for assignments...
      </div>
    )
  }

  return (
    <div className="border border-yellow-600 p-4 font-mono text-sm space-y-3">
      <div className="text-xs text-yellow-700">ACTIVE CONTRACTS</div>
      {active.map(contract => {
        const timeRemaining = getTimeRemaining(contract.deadline)
        const isUrgent = timeRemaining < 15

        const borderColor = isUrgent ? 'border-red-500' : 'border-yellow-600'
        const bgColor = isUrgent ? 'bg-red-950' : ''
        const textColor = isUrgent ? 'text-red-400' : 'text-yellow-400'
        const difficultyColor =
          contract.difficulty === 'hard' ? 'text-yellow-400' : 'text-green-400'

        return (
          <div
            key={contract.id}
            className={`border ${borderColor} ${bgColor} px-3 py-2 space-y-1`}
          >
            <div className="flex justify-between">
              <span className={textColor}>
                {contract.id}{' '}
                <span className={`text-xs ${difficultyColor}`}>
                  [{contract.difficulty.toUpperCase()}]
                </span>
              </span>
              <span className={textColor}>
                {isUrgent ? '⚠ ' : ''}
                {timeRemaining}s ⏱
              </span>
            </div>
            <div className={textColor}>
              {asciiBar(contract.currentVolume, contract.targetVolume)}{' '}
              {contract.currentVolume} / {contract.targetVolume}
            </div>
            <div className="text-xs text-gray-600">
              reward: {contract.reward}¢ &nbsp; penalty: {contract.penalty}¢
            </div>
          </div>
        )
      })}
    </div>
  )
}
```

```typescript
import { Upgrade } from '@/lib/engine/types'

interface UpgradeShopProps {
  upgrades: Upgrade[]
  purchasedUpgradeIds: string[]
  credits: number
  onPurchase: (upgradeId: string) => void
}

function describeEffect(upgrade: Upgrade): string {
  const parts: string[] = []
  if (upgrade.effects.throughput) parts.push(`+${upgrade.effects.throughput} throughput`)
  if (upgrade.effects.efficiency)
    parts.push(`+${upgrade.effects.efficiency.toFixed(1)} efficiency`)
  return parts.join(', ')
}

export function UpgradeShop({
  upgrades,
  purchasedUpgradeIds,
  credits,
  onPurchase,
}: UpgradeShopProps) {
  return (
    <div className="border border-purple-500 p-4 font-mono text-sm space-y-2">
      <div className="text-xs text-purple-700">UPGRADES</div>
      {upgrades.map(upgrade => {
        const owned = purchasedUpgradeIds.includes(upgrade.id)
        const canAfford = credits >= upgrade.cost

        return (
          <div
            key={upgrade.id}
            className="flex justify-between items-center py-1 border-b border-gray-800 last:border-0"
          >
            <div>
              <span className="text-purple-300">{upgrade.name}</span>
              <span className="text-gray-600 text-xs ml-2">
                ({describeEffect(upgrade)})
              </span>
            </div>
            <div className="flex items-center gap-3 shrink-0">
              <span className="text-purple-400">{upgrade.cost}¢</span>
              {owned ? (
                <span className="text-gray-600 text-xs w-12 text-center">[OWNED]</span>
              ) : canAfford ? (
                <button
                  onClick={() => onPurchase(upgrade.id)}
                  className="border border-purple-400 text-purple-400 px-2 py-0.5 text-xs hover:bg-purple-400 hover:text-black transition-colors w-12 text-center"
                >
                  [BUY]
                </button>
              ) : (
                <span className="text-gray-700 text-xs border border-gray-800 px-2 py-0.5 w-12 text-center cursor-not-allowed">
                  [BUY]
                </span>
              )}
            </div>
          </div>
        )
      })}
    </div>
  )
}
```

**Files:**

The tick loop starts once on mount via `useEffect([tick])`. Since `tick` is a stable Zustand function reference, this effect runs exactly once. The game auto-starts without any player input.

`purchasedUpgradeIds` is derived from `state.nodes.flatMap(n => n.upgrades)` — all upgrade IDs owned across all nodes (Phase 1 has one node).

- [ ] **Step 1: Replace the full contents of `app/page.tsx`**

```typescript
'use client'

import { useEffect } from 'react'
import { useGameStore } from '@/store/gameStore'
import { StatusBar } from '@/components/game/StatusBar'
import { NodeCard } from '@/components/game/NodeCard'
import { ContractList } from '@/components/game/ContractList'
import { UpgradeShop } from '@/components/game/UpgradeShop'

export default function GamePage() {
  const { initializeGame, tick, coolNode, purchaseUpgrade, state } = useGameStore()

  useEffect(() => {
    initializeGame()
  }, [initializeGame])

  useEffect(() => {
    const interval = setInterval(tick, 1000)
    return () => clearInterval(interval)
  }, [tick])

  if (!state) {
    return (
      <div className="bg-black text-green-400 font-mono min-h-screen flex items-center justify-center">
        INITIALIZING...
      </div>
    )
  }

  const purchasedUpgradeIds = state.nodes.flatMap(n => n.upgrades)

  return (
    <div className="bg-black min-h-screen p-4 space-y-3 max-w-lg mx-auto">
      <div className="text-green-700 font-mono text-xs text-center py-2 border-b border-gray-900 tracking-widest">
        TERMINAL FLOW v0.1
      </div>

      <StatusBar
        credits={state.player.credits}
        sentiment={state.player.sentiment}
      />

      {state.nodes.map(node => (
        <NodeCard
          key={node.id}
          node={node}
          onCool={() => coolNode(node.id)}
        />
      ))}

      <ContractList contracts={state.contracts} />

      <UpgradeShop
        upgrades={state.upgrades}
        purchasedUpgradeIds={purchasedUpgradeIds}
        credits={state.player.credits}
        onPurchase={upgradeId => purchaseUpgrade(upgradeId, 'node-1')}
      />
    </div>
  )
}
```
